86 void DRV8305_enable(DRV8305_Handle handle)
87 {
88  DRV8305_Obj *obj = (DRV8305_Obj *)handle;
89  static volatile uint16_t enableWaitTimeOut;
90  uint16_t n = 0;
91 
92  // Enable the DRV8305
93  GPIO_setHigh(obj->gpioHandle,obj->gpioNumber);
94 
95  enableWaitTimeOut = 0;
96 
97  // Make sure the FAULT bit is not set during startup
98  while(((DRV8305_readSpi(handle, Address_Status_1) & DRV8305_STATUS01_FAULT_BITS) != 0) && (enableWaitTimeOut < 1000))
99  {
100  if (++enableWaitTimeOut > 999)
101  {
102  obj->enableTimeOut = true;
103  }
104  }
105 
106  // Wait for the DRV8305 to go through start up sequence
107  for(n=0;n<0xffff;n++)
108  asm(" NOP");
109 
110  return;
111 } // end of DRV8305_enable() function
