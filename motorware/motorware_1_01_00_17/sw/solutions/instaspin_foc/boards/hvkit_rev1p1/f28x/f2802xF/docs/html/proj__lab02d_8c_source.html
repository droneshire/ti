128

136

293  }

353 #ifdef DRV8301_SPI

363  HAL_disablePwm(halHandle);
364

367  gMotorVars.Flag_Run_Identify = false;

369  } // end of for(;;) loop

408 void updateGlobalVariables_motor(CTRL_Handle handle)
409 {
410  CTRL_Obj *obj = (CTRL_Obj *)handle;
411  int32_t tmp;
412 
413  // get the speed estimate
414  gMotorVars.Speed_krpm = EST_getSpeed_krpm(obj->estHandle);
415 
416  // get the real time speed reference coming out of the speed trajectory generator
417  gMotorVars.SpeedTraj_krpm = _IQmpy(CTRL_getSpd_int_ref_pu(handle),EST_get_pu_to_krpm_sf(obj->estHandle));
418 
419  // when calling EST_ functions that return a float, and fpu32 is enabled, an integer is needed as a return
420  // so that the compiler reads the returned value from the accumulator instead of fpu32 registers
421  // get the magnetizing current
422  tmp = EST_getIdRated(obj->estHandle);
423  gMotorVars.MagnCurr_A = *((float_t *)&tmp);
424 
425  // get the rotor resistance
426  tmp = EST_getRr_Ohm(obj->estHandle);
427  gMotorVars.Rr_Ohm = *((float_t *)&tmp);
428 
429  // get the stator resistance
430  tmp = EST_getRs_Ohm(obj->estHandle);
431  gMotorVars.Rs_Ohm = *((float_t *)&tmp);
432 
433  // get the stator inductance in the direct coordinate direction
434  tmp = EST_getLs_d_H(obj->estHandle);
435  gMotorVars.Lsd_H = *((float_t *)&tmp);
436 
437  // get the stator inductance in the quadrature coordinate direction
438  tmp = EST_getLs_q_H(obj->estHandle);
439  gMotorVars.Lsq_H = *((float_t *)&tmp);
440 
441  // get the flux in V/Hz in floating point
442  tmp = EST_getFlux_VpHz(obj->estHandle);
443  gMotorVars.Flux_VpHz = *((float_t *)&tmp);
444 
445  // get the controller state
446  gMotorVars.CtrlState = CTRL_getState(handle);
447 
448  // get the estimator state
449  gMotorVars.EstState = EST_getState(obj->estHandle);
450 
451  // Get the DC buss voltage
452  gMotorVars.VdcBus_kV = _IQmpy(gAdcData.dcBus,_IQ(USER_IQ_FULL_SCALE_VOLTAGE_V/1000.0));
453 
454  return;
455 } // end of updateGlobalVariables_motor() function

458 void recalcKpKi(CTRL_Handle handle)
459 {
460  CTRL_Obj *obj = (CTRL_Obj *)handle;
461  EST_State_e EstState = EST_getState(obj->estHandle);
462 
463  if((EST_isMotorIdentified(obj->estHandle) == false) && (EstState == EST_State_Rs))
464  {
465  float_t Lhf = CTRL_getLhf(handle);
466  float_t Rhf = CTRL_getRhf(handle);
467  float_t RhfoverLhf = Rhf/Lhf;
468  _iq Kp = _IQ(0.25*Lhf*USER_IQ_FULL_SCALE_CURRENT_A/(USER_CTRL_PERIOD_sec*USER_IQ_FULL_SCALE_VOLTAGE_V));
469  _iq Ki = _IQ(RhfoverLhf*USER_CTRL_PERIOD_sec);
470 
471  // set Rhf/Lhf
472  CTRL_setRoverL(handle,RhfoverLhf);
473 
474  // set the controller proportional gains
475  CTRL_setKp(handle,CTRL_Type_PID_Id,Kp);
476  CTRL_setKp(handle,CTRL_Type_PID_Iq,Kp);
477 
478  // set the Id controller gains
479  CTRL_setKi(handle,CTRL_Type_PID_Id,Ki);
480  PID_setKi(obj->pidHandle_Id,Ki);
481 
482  // set the Iq controller gains
483  CTRL_setKi(handle,CTRL_Type_PID_Iq,Ki);
484  PID_setKi(obj->pidHandle_Iq,Ki);
485  }
486 
487  return;
488 } // end of recalcKpKi() function
